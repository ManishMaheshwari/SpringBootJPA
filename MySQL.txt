CREATE TABLE CUSTOMERS (
  ID int(11) unsigned NOT NULL AUTO_INCREMENT,
  FIRSTNAME char(255) DEFAULT NULL,
  LASTNAME char(255) DEFAULT NULL,
  PRIMARY KEY (ID)
) ENGINE=InnoDB AUTO_INCREMENT=236 DEFAULT CHARSET=latin1;

CREATE TABLE ORDERS (
  ID int(11) unsigned NOT NULL AUTO_INCREMENT,
  CUST_ID int(11) unsigned NOT NULL,
  DESCRIPTION char(255),
  PRIMARY KEY (ID),
  CONSTRAINT fk_orders_customer FOREIGN KEY (CUST_ID) REFERENCES CUSTOMERS (ID)
) ENGINE=InnoDB AUTO_INCREMENT=236 DEFAULT CHARSET=latin1;

CREATE TABLE ADDRESSES (
  ID int(11) unsigned NOT NULL AUTO_INCREMENT,
  CUST_ID int(11) unsigned NOT NULL,
  ADDRESS char(255),
  PRIMARY KEY (ID),
  CONSTRAINT fk_addresses_customer FOREIGN KEY (CUST_ID) REFERENCES CUSTOMERS (ID)
) ENGINE=InnoDB AUTO_INCREMENT=236 DEFAULT CHARSET=latin1;
